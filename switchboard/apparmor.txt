#include <tunables/global>

profile switchboard flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # The container runs under this named AppArmor profile (mediated — no host
  # escape). We grant broad file/capability/signal/network access because
  # s6-overlay's init plus Asterisk need it across many paths; this can be
  # tightened later. This mirrors the documented Home Assistant add-on pattern.
  file,
  signal,
  capability,
  network,

  # s6-overlay init system: ensure the boot chain can exec.
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # bashio helpers used by the run scripts.
  /usr/lib/bashio/** ix,
  /tmp/** rwk,
}
