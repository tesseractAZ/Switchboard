#include <tunables/global>

profile home_pbx flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Networking: SIP signalling + RTP media + the localhost AMI/Ingress sockets.
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,
  network netlink raw,

  # s6-overlay (init: false → base image's s6 supervises our services).
  /init ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rix,
  /run/{s6,s6-rc*,service}/** rwk,
  /run/** rwk,

  # bashio + busybox helpers used by the run scripts.
  /usr/lib/bashio/** r,
  /usr/bin/bashio rix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,

  # Add-on payload.
  /usr/bin/pbx-generate-config rix,
  /usr/share/pbx/** r,

  # Python runtime for the Ingress UI.
  /usr/bin/python3* rix,
  /usr/lib/python3*/** r,
  /usr/lib/python3*/**.so* mr,

  # Asterisk binary, modules and runtime state.
  /usr/sbin/asterisk rix,
  /usr/lib/asterisk/** mr,
  /usr/share/asterisk/** r,
  /etc/asterisk/** rw,
  /var/lib/asterisk/** rwk,
  /var/log/asterisk/** rwk,
  /var/run/asterisk/** rwk,
  /var/spool/asterisk/** rwk,

  # Add-on persistent config / options.
  /data/** rw,
  /data r,

  # System info Asterisk and Python read at startup.
  /proc/sys/kernel/random/uuid r,
  /proc/*/stat r,
  /proc/*/status r,
  /proc/cpuinfo r,
  /proc/meminfo r,
  /sys/devices/system/cpu/** r,
  /etc/passwd r,
  /etc/group r,
  /etc/nsswitch.conf r,
  /etc/resolv.conf r,
  /etc/hosts r,
  /etc/localtime r,
  /etc/ssl/** r,
}
